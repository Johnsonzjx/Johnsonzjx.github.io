快速排序

把第一个数作为标准，先从后往前找到比该数小的数，在原位置赋值；再从前往后找到比该数大的数，在上一次改变的位置赋新值，如此循环，直到指针交叉

static void quickSort(int[] numArray){
    qSort(numArray, 0, numArray.length-1);
    System.out.println(Arrays.toString(numArray));
}
private static void qSort(int[] a,int low,int high){
    int pivot = 0;
    if(low < high){
        //将数组一分为二
        pivot = partition(a,low,high);
        //对第一部分进行递归排序
        qSort(a,low,pivot);
        //对第二部分进行递归排序
        qSort(a,pivot + 1,high);
    }
}
//partition函数，实现三数中值分割法
private static int partition(int[] a,int low,int high) {
    int pivotkey = a[low];   //选取第一个元素为枢轴记录
    while (low < high) {
        //将比枢轴记录小的交换到低端
        while (low < high && a[high] >= pivotkey) {
            high--;
        }
        //采用替换而不是交换的方式操作
        a[low] = a[high];
        //将比枢轴记录大的交换到高端
        while (low < high && a[low] <= pivotkey) {
            low++;
        }
        a[high] = a[low];
    }
    //枢纽所在位置赋值
    a[low] = pivotkey;
    //返回枢纽所在的位置
    return low;
}

桶排序

    public static void bucketSort(int[] data, int min, int max) {
        // 缓存数组
        int[] tmp = new int[data.length];
        // buckets用于记录待排序元素的信息
        // buckets数组定义了max-min个桶
        int[] buckets = new int[max - min +  1];
        // 计算每个元素在序列出现的次数
        for (int i = 0; i < data.length; i++) {
            buckets[data[i] - min]++;
        }
        // 计算“落入”各桶内的元素在有序序列中的位置
        for (int i = 1; i < max - min; i++) {
            buckets[i] = buckets[i] + buckets[i - 1];
        }
        // 将data中的元素完全复制到tmp数组中
        System.arraycopy(data, 0, tmp, 0, data.length);
        // 根据buckets数组中的信息将待排序列的各元素放入相应位置
        for (int k = data.length - 1; k >= 0; k--) {
            data[--buckets[tmp[k] - min]] = tmp[k];
        }
        System.out.println(Arrays.toString(data));
    }

归并排序

将数组进行不断递归细分，对每一个子部分进行排序，最后再进行合并

public static void andSort(int[] numArray){
        int[] temp = new int[numArray.length];
        divide(numArray, temp, 0, numArray.length - 1);
        System.out.println(Arrays.toString(numArray));

    }
    private static void divide(int[] array, int[] temp, int left, int right){
        if(left < right){
            int mid = (right + left) /  2;
            divide(array, temp, left, mid);
            divide(array, temp, mid+1, right);
            merge(array, temp, left, mid+1, right);
        }
    }
    private static void merge(int[] array, int[] temp, int leftStart, int rightStart, int rightEnd){
        int leftEnd = rightStart - 1;
        int eleNum = rightEnd - leftStart + 1;
        int tempPos = leftStart;
        while(leftStart <= leftEnd && rightStart <= rightEnd){
            if(array[leftStart] <= array[rightStart]){
                temp[tempPos++] = array[leftStart++];
            }
            else{
                temp[tempPos++] = array[rightStart++];
            }
        }
        while(leftStart <= leftEnd){
            temp[tempPos++] = array[leftStart++];
        }
        while(rightStart <= rightEnd){
            temp[tempPos++] = array[rightStart++];
        }
        for(int i=0;i<eleNum;i++) {
            array[rightEnd] = temp[rightEnd];
            rightEnd--;
        }
    }
